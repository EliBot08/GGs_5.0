# Manual GitHub Push Commands (If batch script fails)
# Run these commands in Command Prompt or PowerShell

# Step 1: Initialize and configure
git init
git config user.name "GGs Developer"
git config user.email "ggs@example.com"
git remote add origin https://github.com/EliBot08/GGs_5.0.git

# Step 2: Add .gitignore first
git add .gitignore
git commit -m "Add comprehensive .gitignore for .NET project"

# Batch 1: Core files
git add GGs.sln Directory.Build.targets
git commit -m "Batch 1: Core solution files"
git push origin main

# Batch 2: Source code
git add agents/ clients/ server/ shared/ src/
git commit -m "Batch 2: Core application source code"
git push origin main

# Batch 3: Tests
git add tests/
git commit -m "Batch 3: Test projects"
git push origin main

# Batch 4: Documentation and tools
git add docs/ packaging/ tools/ *.md *.bat *.ps1 *.cmd *.txt *.html *.vbs
git commit -m "Batch 4: Documentation and build scripts"
git push origin main

# Batch 5: Configuration files
git add *.json *.xml *.config .github/
git commit -m "Batch 5: Configuration and GitHub workflows"
git push origin main

# Verification
git status
git log --oneline -10

# Check repository size
git count-objects -vH
